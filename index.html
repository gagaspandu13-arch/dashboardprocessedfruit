/**
 * Cut Fruit & Juice Operational Dashboard - Google Apps Script Backend (v2.2)
 * Includes Automatic Logging to "ABSENSI AC" Sheet
 */
const SS_CONFIG = {
    SHEET_PROCESSED_NAMES: ["PROCESSED FRUIT", "PROCESSED", "CUT FRUIT", "JUICE"],
    SHEET_RAW_NAMES: ["RAW", "RAW FRUIT", "BAHAN MENTAH", "RAW MATERIAL"],
    SHEET_MAPPING_NAMES: ["MAPPING PLU", "MAPPING", "MAP PLU", "PLU MAPPING"],
    SHEET_ABSENSI_AC: "ABSENSI AC",
    SHEET_CACHE: "_SUMMARY_CACHE_",
    CHUNK_SIZE: 35000
};

function doGet(e) {
    try {
        const ss = SpreadsheetApp.getActiveSpreadsheet();

        // Handle AC Attendance Logging via GET or POST
        if (e && e.parameter && e.parameter.action === "absen_ac") {
            return recordAcAttendance(e.parameter.acName);
        }

        let cacheSheet = ss.getSheetByName(SS_CONFIG.SHEET_CACHE);
        const forceRefresh = e && e.parameter && (e.parameter.refresh === "true" || e.parameter.force === "1");

        if (forceRefresh || !cacheSheet || cacheSheet.getLastRow() < 2) {
            buildDashboardCache();
            cacheSheet = ss.getSheetByName(SS_CONFIG.SHEET_CACHE);
        }

        const lastRow = cacheSheet ? cacheSheet.getLastRow() : 0;
        let rawData = "";
        if (lastRow >= 2) {
            rawData = cacheSheet.getRange(2, 1, lastRow - 1, 1).getValues().map(r => r[0]).join('');
        }

        if (!rawData) {
            buildDashboardCache();
            cacheSheet = ss.getSheetByName(SS_CONFIG.SHEET_CACHE);
            const retryLastRow = cacheSheet ? cacheSheet.getLastRow() : 0;
            if (retryLastRow >= 2) {
                rawData = cacheSheet.getRange(2, 1, retryLastRow - 1, 1).getValues().map(r => r[0]).join('');
            }
        }

        return ContentService.createTextOutput(rawData)
            .setMimeType(ContentService.MimeType.JSON);
    } catch (err) {
        return ContentService.createTextOutput(JSON.stringify({
            status: "error", message: err.toString(), stack: err.stack
        })).setMimeType(ContentService.MimeType.JSON);
    }
}

function doPost(e) {
    try {
        let postData = {};
        if (e && e.postData && e.postData.contents) {
            try { postData = JSON.parse(e.postData.contents); } catch(ex){}
        }
        const action = postData.action || (e && e.parameter ? e.parameter.action : "");
        const acName = postData.acName || (e && e.parameter ? e.parameter.acName : "");

        if (action === "absen_ac") {
            return recordAcAttendance(acName);
        }
        return doGet(e);
    } catch (err) {
        return ContentService.createTextOutput(JSON.stringify({
            status: "error", message: err.toString()
        })).setMimeType(ContentService.MimeType.JSON);
    }
}

function recordAcAttendance(acName) {
    if (!acName) {
        return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "acName is required" })).setMimeType(ContentService.MimeType.JSON);
    }

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SS_CONFIG.SHEET_ABSENSI_AC);
    
    if (!sheet) {
        sheet = ss.insertSheet(SS_CONFIG.SHEET_ABSENSI_AC);
        sheet.appendRow(["WAKTU & TANGGAL", "NAMA AC"]);
        sheet.getRange(1, 1, 1, 2).setFontWeight("bold").setBackground("#d1fae5");
    }

    const timeStr = Utilities.formatDate(new Date(), "Asia/Jakarta", "yyyy-MM-dd HH:mm:ss");
    sheet.appendRow([timeStr, acName]);

    return ContentService.createTextOutput(JSON.stringify({
        status: "success",
        message: "Absensi AC berhasil dicatat ke sheet ABSENSI AC",
        timestamp: timeStr,
        acName: acName
    })).setMimeType(ContentService.MimeType.JSON);
}
